Írjon programot, amely eldönti egy számról, hogy az páros vagy páratlan! 
DECLARE
    n NUMBER := 127; -- n NUMBER := &n;

BEGIN
    IF n mod 2 = 0 THEN
        DBMS_OUTPUT.PUT_LINE('The number ' || TO_CHAR(n) || ' is even.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('The number ' || TO_CHAR(n) || ' is odd.');
    END IF;

END;
Írjon programot, amely megadja, hogy három szám közül melyik a legnagyobb! 
DECLARE
    a NUMBER := 100;
    b NUMBER := 25;
    c NUMBER := 50;

BEGIN
    DBMS_OUTPUT.PUT_LINE('A=' || TO_CHAR(a) || ' B=' || TO_CHAR(b) || ' C=' || TO_CHAR(c));

    IF a > b AND a > c THEN
        DBMS_OUTPUT.PUT_LINE('A is the greatest number.');
    ELSIF b > a AND b > c THEN
        DBMS_OUTPUT.PUT_LINE('B is the greatest number.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('C is the greatest number.');
    END IF;

END;
Írjon programot, amely meghatározza, hogy egy adott év szökőév-e! 
DECLARE
    v_year NUMBER := 2020; -- year NUMBER := &year;

BEGIN
    IF v_year mod 4 = 0 AND v_year mod 100 != 0 OR v_year mod 400 = 0 THEN
        DBMS_OUTPUT.PUT_LINE(TO_CHAR(v_year) || ' is a leap year.');
    ELSE
        DBMS_OUTPUT.PUT_LINE(TO_CHAR(v_year) || ' is not a leap year.');
    END IF;

END;
Írjon programot, amely előállítja az első n darab Fibonacci számot! 
DECLARE
    f1 NUMBER := 0;
    f2 NUMBER := 1;
    f_temp NUMBER;
    n NUMBER := 10;

BEGIN
    DBMS_OUTPUT.PUT_LINE('The first ' || TO_CHAR(n) || ' Fibonacci numbers:');

    FOR I IN 1..n
    LOOP
        IF I = 1 THEN
            DBMS_OUTPUT.PUT_LINE(TO_CHAR(f1));
        ELSIF I = 2 THEN
            DBMS_OUTPUT.PUT_LINE(TO_CHAR(f2));
        ELSE
            f_temp := f1 + f2;
            f1 := f2;
            f2 := f_temp;
            DBMS_OUTPUT.PUT_LINE(TO_CHAR(f2));
        END IF;
    END LOOP;

END;
Írjon programot, amely eldönti egy számról, hogy az prím-e! 
DECLARE
    n NUMBER := 127;
    prime BOOLEAN := TRUE;

BEGIN

    FOR i IN 2..SQRT(n)
    LOOP
        IF n mod i = 0 THEN
            prime := FALSE;
            EXIT;
        END IF;
    END LOOP;

    IF prime THEN
        DBMS_OUTPUT.PUT_LINE(TO_CHAR(n) || ' is prime.');
    ELSE
        DBMS_OUTPUT.PUT_LINE(TO_CHAR(n) || ' is not prime.');
    END IF;

END;
Írjon programot, amely eldönti egy számról, hogy az Armstrong szám-e! 
DECLARE
    v_num NUMBER := 92727;
    v_sum NUMBER := 0;
    v_rem NUMBER;
    v_len NUMBER;
    v_tmp NUMBER;

BEGIN

    v_len := LENGTH(TO_CHAR(v_num));
    v_tmp := v_num;

    WHILE v_tmp > 0
    LOOP
        v_rem := v_tmp mod 10;
        v_sum := v_sum + v_rem**v_len;
        v_tmp := TRUNC(v_tmp / 10);
    END LOOP;

    IF v_num = v_sum THEN
        DBMS_OUTPUT.PUT_LINE(TO_CHAR(v_num) || ' is an Armstrong number.');
    ELSE
        DBMS_OUTPUT.PUT_LINE(TO_CHAR(v_num) || ' is not an Armstrong number.');
    END IF;

END;
