CREATE OR REPLACE FUNCTION salary_rank(p_emp_id EMPLOYEES.EMPLOYEE_ID%TYPE)
RETURN NUMBER IS
    v_salary EMPLOYEES.SALARY%TYPE;
    v_job_id JOBS.JOB_ID%TYPE;
    v_rank NUMBER;

BEGIN
    SELECT SALARY, JOB_ID INTO v_salary, v_job_id FROM EMPLOYEES
    WHERE EMPLOYEE_ID = p_emp_id;

    SELECT COUNT(DISTINCT SALARY) + 1 INTO v_rank FROM EMPLOYEES
    WHERE JOB_ID = v_job_id and SALARY > v_salary;

    RETURN v_rank;

    EXCEPTION
        WHEN NO_DATA_FOUND THEN RETURN NULL;

END salary_rank;
/

SELECT FIRST_NAME || ' ' || LAST_NAME "Name", salary_rank(EMPLOYEE_ID) "Salary rank" FROM EMPLOYEES
WHERE EMPLOYEE_ID = 154;

CREATE OR REPLACE FUNCTION topt_job_title(p_dept_id DEPARTMENTS.DEPARTMENT_ID%TYPE)
RETURN JOBS.JOB_TITLE%TYPE IS
    TYPE t_job_ids IS TABLE OF JOBS.JOB_ID%TYPE;
    v_job_ids t_job_ids;
    v_job_title JOBS.JOB_TITLE%TYPE;

BEGIN
    SELECT JOB_ID BULK COLLECT INTO v_job_ids FROM EMPLOYEES
    WHERE DEPARTMENT_ID = p_dept_id GROUP BY JOB_ID ORDER BY COUNT(*) DESC;

    IF v_job_ids.COUNT = 0 THEN
        RETURN 'N/A';
    ELSE
        SELECT JOB_TITLE INTO v_job_title FROM JOBS WHERE JOB_ID = v_job_ids(1);
        RETURN v_job_title;
    END IF;

END topt_job_title;
/

SELECT DEPARTMENT_NAME, topt_job_title(DEPARTMENT_ID) FROM DEPARTMENTS ORDER BY DEPARTMENT_NAME;
CREATE OR REPLACE FUNCTION topt_job_title(p_dept_id DEPARTMENTS.DEPARTMENT_ID%TYPE)
RETURN JOBS.JOB_TITLE%TYPE IS
    v_num_emps NUMBER;
    v_job_title JOBS.JOB_TITLE%TYPE;

BEGIN
    SELECT COUNT(*) INTO v_num_emps FROM EMPLOYEES WHERE DEPARTMENT_ID = p_dept_id;

    IF v_num_emps = 0 THEN
        RETURN 'N/A';
    ELSE
        SELECT JOB_TITLE INTO v_job_title FROM EMPLOYEES JOIN JOBS USING (JOB_ID)
        WHERE DEPARTMENT_ID = p_dept_id GROUP BY JOB_TITLE ORDER BY COUNT(*) DESC
        FETCH FIRST ROW ONLY;
        RETURN v_job_title;
    END IF;

END topt_job_title;
/

SELECT DEPARTMENT_NAME, topt_job_title(DEPARTMENT_ID) FROM DEPARTMENTS ORDER BY DEPARTMENT_NAME;

DECLARE
    TYPE t_rec IS RECORD(avg_sal NUMBER, emp_cnt NUMBER);
    v_man_rec t_rec;
    v_fn VARCHAR2(100);

    FUNCTION man_emp_info(p_man_id EMPLOYEES.EMPLOYEE_ID%TYPE)
    RETURN t_rec IS
        v_rec t_rec;
    BEGIN
        SELECT AVG(SALARY), COUNT(EMPLOYEE_ID) INTO v_rec FROM EMPLOYEES
        WHERE MANAGER_ID = p_man_id;
        RETURN v_rec;
    END man_emp_info;

BEGIN
    FOR c IN (SELECT DISTINCT MANAGER_ID FROM EMPLOYEES WHERE MANAGER_ID IS NOT NULL ORDER BY MANAGER_ID)
    LOOP
        SELECT FIRST_NAME || ' ' || LAST_NAME INTO v_fn FROM EMPLOYEES
        WHERE EMPLOYEE_ID = c.MANAGER_ID;
        v_man_rec := man_emp_info(c.MANAGER_ID);
        IF v_man_rec.avg_sal > 8000 THEN
            DBMS_OUTPUT.PUT_LINE(v_fn || ' (' || v_man_rec.emp_cnt || '): ' || ROUND(v_man_rec.avg_sal));
        END IF;
    END LOOP;

END;
4. feladat

Hozzon létre egy DEPT_STATS nevű táblát! Ennek oszlopai:
- DEPT_ID: NUMBER(4) típusú, elsődleges kulcs
- NUM_EMP: NUMBER(3) típusú
- AVG_EXP: NUMBER(3,1) típusú

Írjon lokális eljárást, amely paraméterként egy részleg azonosítót kap, és beszúr egy új sort a részleg azonosítójával, az adott részlegen dolgozók számával és a dolgozók átlagos munkatapasztalatával (években, egy tizedes pontossággal) a DEPT_STATS táblába, ha a részleg még nem szerepel a táblában!
Írjon programot, amely feltölti a DEPT_STATS táblát azon részlegek adataival, amelyekhez van megadva dolgozó az adatbázisban!
CREATE TABLE DEPT_STATS(DEPT_ID NUMBER(4) PRIMARY KEY,
                        NUM_EMP NUMBER(3),
                        AVG_EXP NUMBER(3,1));
/

DECLARE
    PROCEDURE dept_ins(p_dept_id DEPARTMENTS.DEPARTMENT_ID%TYPE) IS
        v_avg_exp NUMBER;
        v_count NUMBER;
    BEGIN
        SELECT COUNT(*) INTO v_count FROM DEPT_STATS WHERE DEPT_ID = p_dept_id;
        IF v_count = 0 THEN
            SELECT AVG(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)/12), COUNT(*) INTO v_avg_exp, v_count FROM EMPLOYEES
            WHERE DEPARTMENT_ID = p_dept_id;
            INSERT INTO DEPT_STATS VALUES(p_dept_id, v_count, v_avg_exp);
        END IF;
    END dept_ins;

BEGIN
    FOR c IN (SELECT DISTINCT DEPARTMENT_ID FROM EMPLOYEES WHERE DEPARTMENT_ID IS NOT NULL)
    LOOP
        dept_ins(c.DEPARTMENT_ID);
    END LOOP;

END;

DECLARE
    PROCEDURE dept_ins(p_dept_id DEPARTMENTS.DEPARTMENT_ID%TYPE) IS
        v_count NUMBER(1);
    BEGIN
        SELECT COUNT(*) INTO v_count FROM DEPT_STATS WHERE DEPT_ID = p_dept_id AND ROWNUM = 1;
        IF v_count = 0 THEN
            INSERT INTO DEPT_STATS
            SELECT p_dept_id, COUNT(*), ROUND(AVG(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)/12),1) FROM EMPLOYEES
            WHERE DEPARTMENT_ID = p_dept_id;
        END IF;
    END dept_ins;

BEGIN
    FOR c IN (SELECT DISTINCT DEPARTMENT_ID FROM EMPLOYEES WHERE DEPARTMENT_ID IS NOT NULL)
    LOOP
        dept_ins(c.DEPARTMENT_ID);
    END LOOP;
